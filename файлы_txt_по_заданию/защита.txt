Наличие пароля для root  и других пользователей  
Наличие безопасного подключения к серверу через SSH
Наличие настреоенного фаервола iptables , который будет регулировать подключение к серверу.

Если в течении 2 минут пароль будет введён неправильно 7 раз, этот ip будет заблокирован на 12 часов.  

[DEFAULT]
ignoreip = 95.111.123.21
[sshd]  
enabled  = true  
findtime = 120  
maxretry = 3  
bantime = 43200
